C Program - Morse Code Converter

#include <stdio.h>
#include <string.h>

char* morse[] = { ".-", "-...", "-.-.", "-..", ".", "..-.", "--.",
                  "....", "..", ".---", "-.-", ".-..", "--", "-.",
                  "---", ".--.", "--.-", ".-.", "...", "-", "..-",
                  "...-", ".--", "-..-", "-.--", "--..", "-----", 
                  ".----", "..---", "...--", "....-", ".....", 
                  "-....", "--...", "---..", "----." };

char letters[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

void toMorse(char* text) {
    for(int i = 0; text[i]; i++) {
        char ch = toupper(text[i]);
        if(ch == ' ') printf("/ ");
        else {
            char* ptr = strchr(letters, ch);
            if(ptr) printf("%s ", morse[ptr - letters]);
        }
    }
}

int main() {
    char input[100];
    printf("Enter text: ");
    gets(input);
    printf("Morse Code: ");
    toMorse(input);
    return 0;
}
